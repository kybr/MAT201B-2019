``` {#lst:vec}
  // we can do arithmetic with vectors
  Vec3f a(1, 0, 0), b(0, 1, 0);
  cout << "a:" << a << endl;
  cout << "b:" << b << endl;
  cout << (a + b) << endl;

  // this includes scaling
  cout << "a*0.1:" << a * 0.1 << endl;
  cout << "(a+b)*0.1:" << (a + b) * 0.1 << endl;

  for (float t = 0; t < 1; t += 0.099)
    cout << "t:" << a * (1 - t) + b * t << endl;
```
